Main Menu-
	Enter = Start Game
	H = High Score
	Esc = Quit

Game Play-
	Space bar = Fire
	Arrow Keys = Movement
	Brown plane = 10 points
	Grey plane = 15 points
	Blue plane = 20 points
	Greeen plane = 25 points
	E = Main Menu

High Score-
	E = Main Menu
	Enter = Start Game
